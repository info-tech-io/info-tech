## –®–∞–≥ 5.2: –°–æ–∑–¥–∞–Ω–∏–µ –∫–æ–º–∞–Ω–¥—ã `add`

**–¶–µ–ª—å:** –†–µ–∞–ª–∏–∑–æ–≤–∞—Ç—å –∫–æ–º–∞–Ω–¥—É –¥–ª—è –ø–æ–¥–∫–ª—é—á–µ–Ω–∏—è –º–æ–¥—É–ª–µ–π –∫ –ø–ª–∞—Ç—Ñ–æ—Ä–º–µ

**–§–∞–π–ª info_tech_cli/commands/add.py:**
```python
"""
Add module command - –ø–æ–¥–∫–ª—é—á–µ–Ω–∏–µ —Å—É—â–µ—Å—Ç–≤—É—é—â–µ–≥–æ –º–æ–¥—É–ª—è –∫ –ø–ª–∞—Ç—Ñ–æ—Ä–º–µ InfoTech.io
"""

import os
import json
import click
import requests
from datetime import datetime
from pathlib import Path
import tempfile
import subprocess
import yaml

class ModuleAdder:
    def __init__(self, github_token, github_org):
        self.github_token = github_token
        self.github_org = github_org
        self.github_api = "https://api.github.com"
        self.session = requests.Session()
        self.session.headers.update({
            "Authorization": f"token {github_token}",
            "Accept": "application/vnd.github.v3+json",
            "User-Agent": "InfoTech-CLI/1.0"
        })

    def add_module(self, source_repo, module_name=None, interactive=False):
        """–ì–ª–∞–≤–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è –¥–æ–±–∞–≤–ª–µ–Ω–∏—è –º–æ–¥—É–ª—è"""
        
        click.echo(f"üöÄ Adding module from: {source_repo}")
        
        # 1. –ü–∞—Ä—Å–∏–Ω–≥ –∏ –≤–∞–ª–∏–¥–∞—Ü–∏—è –≤—Ö–æ–¥–Ω–æ–≥–æ —Ä–µ–ø–æ–∑–∏—Ç–æ—Ä–∏—è
        repo_info = self.parse_repository(source_repo)
        if not repo_info:
            raise click.ClickException("‚ùå Invalid repository format")
        
        # 2. –û–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ –∏–º–µ–Ω–∏ –º–æ–¥—É–ª—è
        if not module_name:
            module_name = self.extract_module_name(repo_info, interactive)
        
        # 3. –ü—Ä–æ–≤–µ—Ä–∫–∞ —Ä–µ–ø–æ–∑–∏—Ç–æ—Ä–∏—è
        if not self.validate_source_repository(repo_info):
            raise click.ClickException("‚ùå Source repository validation failed")
        
        # 4. –°–æ–∑–¥–∞–Ω–∏–µ —Ñ–æ—Ä–∫–∞ –µ—Å–ª–∏ –Ω—É–∂–Ω–æ
        target_repo = self.ensure_fork_in_organization(repo_info)
        
        # 5. –î–æ–±–∞–≤–ª–µ–Ω–∏–µ workflow –≤ —Ä–µ–ø–æ–∑–∏—Ç–æ—Ä–∏–π –º–æ–¥—É–ª—è
        self.add_platform_workflow(target_repo)
        
        # 6. –û–±–Ω–æ–≤–ª–µ–Ω–∏–µ modules.json
        self.update_modules_registry(module_name, target_repo, repo_info)
        
        # 7. –ó–∞–ø—É—Å–∫ –ø–µ—Ä–≤–∏—á–Ω–æ–≥–æ –±–∏–ª–¥–∞
        self.trigger_initial_build(module_name, target_repo)
        
        click.echo(f"‚úÖ Module '{module_name}' successfully added to platform!")
        click.echo(f"üåê Will be available at: https://{module_name.replace('_', '-')}.infotecha.ru")

    def parse_repository(self, source_repo):
        """–ü–∞—Ä—Å–∏–Ω–≥ —Ä–∞–∑–ª–∏—á–Ω—ã—Ö —Ñ–æ—Ä–º–∞—Ç–æ–≤ —Ä–µ–ø–æ–∑–∏—Ç–æ—Ä–∏—è"""
        
        # –ü–æ–ª–Ω—ã–π URL
        if source_repo.startswith("https://github.com/"):
            parts = source_repo.replace("https://github.com/", "").strip("/").split("/")
            if len(parts) >= 2:
                return {"owner": parts[0], "name": parts[1]}
        
        # –ö–æ—Ä–æ—Ç–∫–∏–π —Ñ–æ—Ä–º–∞—Ç owner/repo
        elif "/" in source_repo:
            parts = source_repo.split("/")
            if len(parts) == 2:
                return {"owner": parts[0], "name": parts[1]}
        
        # –¢–æ–ª—å–∫–æ –∏–º—è —Ä–µ–ø–æ–∑–∏—Ç–æ—Ä–∏—è (–ø—Ä–µ–¥–ø–æ–ª–∞–≥–∞–µ–º –≤ –Ω–∞—à–µ–π –æ—Ä–≥–∞–Ω–∏–∑–∞—Ü–∏–∏)
        elif source_repo.startswith("mod_"):
            return {"owner": self.github_org, "name": source_repo}
        
        return None

    def extract_module_name(self, repo_info, interactive):
        """–ò–∑–≤–ª–µ—á–µ–Ω–∏–µ –∏–º–µ–Ω–∏ –º–æ–¥—É–ª—è"""
        
        # –ê–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–æ–µ –∏–∑–≤–ª–µ—á–µ–Ω–∏–µ –∏–∑ –∏–º–µ–Ω–∏ —Ä–µ–ø–æ–∑–∏—Ç–æ—Ä–∏—è
        repo_name = repo_info["name"]
        if repo_name.startswith("mod_"):
            suggested_name = repo_name[4:]  # –£–±–∏—Ä–∞–µ–º –ø—Ä–µ—Ñ–∏–∫—Å mod_
        else:
            suggested_name = repo_name
        
        if interactive:
            return click.prompt(
                f"Module name", 
                default=suggested_name,
                type=str
            )
        else:
            click.echo(f"üìù Using module name: {suggested_name}")
            return suggested_name

    def validate_source_repository(self, repo_info):
        """–í–∞–ª–∏–¥–∞—Ü–∏—è –∏—Å—Ö–æ–¥–Ω–æ–≥–æ —Ä–µ–ø–æ–∑–∏—Ç–æ—Ä–∏—è"""
        
        click.echo("üîç Validating source repository...")
        
        # –ü—Ä–æ–≤–µ—Ä–∫–∞ —Å—É—â–µ—Å—Ç–≤–æ–≤–∞–Ω–∏—è —Ä–µ–ø–æ–∑–∏—Ç–æ—Ä–∏—è
        repo_url = f"{self.github_api}/repos/{repo_info['owner']}/{repo_info['name']}"
        response = self.session.get(repo_url)
        
        if response.status_code != 200:
            click.echo(f"‚ùå Repository not found or not accessible")
            return False
        
        # –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞–ª–∏—á–∏—è content/ –ø–∞–ø–∫–∏
        contents_url = f"{self.github_api}/repos/{repo_info['owner']}/{repo_info['name']}/contents"
        response = self.session.get(contents_url)
        
        if response.status_code == 200:
            contents = response.json()
            has_content_dir = any(
                item["name"] == "content" and item["type"] == "dir" 
                for item in contents
            )
            
            if not has_content_dir:
                click.echo("‚ö†Ô∏è  Warning: No 'content/' directory found in repository")
                if not click.confirm("Continue anyway?"):
                    return False
        
        click.echo("‚úÖ Repository validation passed")
        return True

    def ensure_fork_in_organization(self, repo_info):
        """–°–æ–∑–¥–∞–Ω–∏–µ —Ñ–æ—Ä–∫–∞ –≤ –æ—Ä–≥–∞–Ω–∏–∑–∞—Ü–∏–∏ –µ—Å–ª–∏ –Ω—É–∂–Ω–æ"""
        
        # –ï—Å–ª–∏ —Ä–µ–ø–æ–∑–∏—Ç–æ—Ä–∏–π —É–∂–µ –≤ –Ω–∞—à–µ–π –æ—Ä–≥–∞–Ω–∏–∑–∞—Ü–∏–∏
        if repo_info["owner"] == self.github_org:
            click.echo(f"‚úÖ Repository already in {self.github_org} organization")
            return repo_info
        
        click.echo(f"üç¥ Creating fork in {self.github_org} organization...")
        
        # –ü—Ä–æ–≤–µ—Ä—è–µ–º, –Ω–µ—Ç –ª–∏ —É–∂–µ —Ñ–æ—Ä–∫–∞
        fork_name = repo_info["name"]
        fork_url = f"{self.github_api}/repos/{self.github_org}/{fork_name}"
        response = self.session.get(fork_url)
        
        if response.status_code == 200:
            click.echo("‚úÖ Fork already exists")
            return {"owner": self.github_org, "name": fork_name}
        
        # –°–æ–∑–¥–∞–µ–º —Ñ–æ—Ä–∫
        fork_url = f"{self.github_api}/repos/{repo_info['owner']}/{repo_info['name']}/forks"
        fork_data = {"organization": self.github_org}
        
        response = self.session.post(fork_url, json=fork_data)
        
        if response.status_code == 202:
            click.echo("‚úÖ Fork created successfully")
            return {"owner": self.github_org, "name": fork_name}
        else:
            raise click.ClickException(f"‚ùå Failed to create fork: {response.text}")

    def add_platform_workflow(self, repo_info):
        """–î–æ–±–∞–≤–ª–µ–Ω–∏–µ workflow –¥–ª—è —É–≤–µ–¥–æ–º–ª–µ–Ω–∏—è –ø–ª–∞—Ç—Ñ–æ—Ä–º—ã"""
        
        click.echo("‚öôÔ∏è Adding platform notification workflow...")
        
        workflow_content = {
            "name": "Notify Platform on Content Update",
            "on": {
                "push": {
                    "branches": ["main"],
                    "paths": ["content/**"]
                }
            },
            "jobs": {
                "notify-platform": {
                    "runs-on": "ubuntu-latest",
                    "steps": [
                        {
                            "name": "Notify InfoTech.io Platform",
                            "uses": "peter-evans/repository-dispatch@v3",
                            "with": {
                                "token": "${{ secrets.PLATFORM_TOKEN }}",
                                "repository": "info-tech-io/infotecha", 
                                "event-type": "module-updated",
                                "client-payload": json.dumps({
                                    "module_name": "${MODULE_NAME}",  # –ë—É–¥–µ—Ç –∑–∞–º–µ–Ω–µ–Ω–æ
                                    "content_repo": repo_info["name"],
                                    "updated_at": "${{ github.event.head_commit.timestamp }}"
                                }, separators=(',', ':'))
                            }
                        }
                    ]
                }
            }
```
