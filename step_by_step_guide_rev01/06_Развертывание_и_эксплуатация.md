        # Remove module directory
        rm -rf "$module_dir"
        log "Removed directory: $module_dir"
    else
        log "Directory does not exist: $module_dir"
    fi

#!/bin/bash

# InfoTech.io Deployment Script

set -e

WEBROOT="/var/www/infotecha.ru"
BACKUP_DIR="/var/www/infotecha.ru/backups"
LOG_FILE="/var/www/infotecha.ru/logs/deployment.log"

# Logging function
log() {
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1" | tee -a "$LOG_FILE"
}

# Error handler
error_handler() {
    log "‚ùå ERROR: Deployment failed at line $1"
    exit 1

trap 'error_handler $LINENO' ERR

# Function: Deploy hub (main page and modules.json)
deploy_hub() {
    log "üöÄ Starting hub deployment..."
    
    # Create backup of current files
    if [ -f "$WEBROOT/index.html" ]; then
        local backup_name="hub-backup-$(date +%Y%m%d_%H%M%S)"
        mkdir -p "$BACKUP_DIR/$backup_name"
        cp "$WEBROOT/index.html" "$BACKUP_DIR/$backup_name/" 2>/dev/null || true
        cp "$WEBROOT/modules.json" "$BACKUP_DIR/$backup_name/" 2>/dev/null || true
        cp "$WEBROOT"/*.css "$BACKUP_DIR/$backup_name/" 2>/dev/null || true
        cp "$WEBROOT"/*.js "$BACKUP_DIR/$backup_name/" 2>/dev/null || true
        log "üì¶ Backup created: $backup_name"
    fi
    
    # Deploy new files (files should be uploaded to $TEMP_DIR by GitHub Actions)
    if [ -d "$TEMP_DIR" ]; then
        log "üìÅ Deploying files from: $TEMP_DIR"
        
        # Copy files with proper permissions
        find "$TEMP_DIR" -type f -name "*.html" -exec cp {} "$WEBROOT/" \;
        find "$TEMP_DIR" -type f -name "*.css" -exec cp {} "$WEBROOT/" \;
        find "$TEMP_DIR" -type f -name "*.js" -exec cp {} "$WEBROOT/" \;
        find "$TEMP_DIR" -type f -name "modules.json" -exec cp {} "$WEBROOT/" \;
        
        # Copy images directory if exists
        if [ -d "$TEMP_DIR/images" ]; then
            cp -r "$TEMP_DIR/images" "$WEBROOT/" 2>/dev/null || true
        fi
        
        # Set proper permissions
        chown -R www-data:www-data "$WEBROOT"/*.html "$WEBROOT"/*.css "$WEBROOT"/*.js "$WEBROOT"/modules.json 2>/dev/null || true
        chown -R www-data:www-data "$WEBROOT/images" 2>/dev/null || true
        chmod 644 "$WEBROOT"/*.html "$WEBROOT"/*.css "$WEBROOT"/*.js "$WEBROOT"/modules.json 2>/dev/null || true
        
        log "‚úÖ Hub files deployed successfully"
    else
        log "‚ö†Ô∏è No temp directory found, skipping file deployment"
    fi
    
    # Validate deployment
    if [ -f "$WEBROOT/index.html" ] && [ -f "$WEBROOT/modules.json" ]; then
        log "‚úÖ Hub deployment validation passed"
    else
        log "‚ùå Hub deployment validation failed"
        return 1
    fi
}

# Function: Deploy specific module
deploy_module() {
    local module_name="$1"
    local module_subdir="$2"
    
    if [ -z "$module_name" ] || [ -z "$module_subdir" ]; then
        log "‚ùå Module name and subdirectory required"
        return 1
    fi
    
    log "üöÄ Starting module deployment: $module_name"
    
    local module_dir="$WEBROOT/$module_subdir"
    
    # Create module directory if doesn't exist
    mkdir -p "$module_dir"
    
    # Create backup
    if [ "$(ls -A $module_dir 2>/dev/null)" ]; then
        local backup_name="module-${module_name}-backup-$(date +%Y%m%d_%H%M%S)"
        mkdir -p "$BACKUP_DIR/$backup_name"
        cp -r "$module_dir"/* "$BACKUP_DIR/$backup_name/" 2>/dev/null || true
        log "üì¶ Module backup created: $backup_name"
    fi
    
    # Deploy module files
    local module_temp="$TEMP_DIR/$module_name"
    if [ -d "$module_temp" ]; then
        log "üìÅ Deploying module from: $module_temp"
        
        # Clear existing content and deploy new
        rm -rf "$module_dir"/*
        cp -r "$module_temp"/* "$module_dir/"
        
        # Set permissions
        chown -R www-data:www-data "$module_dir"
        find "$module_dir" -type f -exec chmod 644 {} \;
        find "$module_dir" -type d -exec chmod 755 {} \;
        
        log "‚úÖ Module $module_name deployed successfully"
    else
        log "‚ùå Module source directory not found: $module_temp"
        return 1
    fi
    
    # Validate module deployment
    if [ -f "$module_dir/index.html" ]; then
        log "‚úÖ Module deployment validation passed"
    else
        log "‚ùå Module deployment validation failed - no index.html"
        return 1
    fi
}

# Function: Cleanup
cleanup() {
    log "üßπ Cleaning up temporary files..."
    rm -rf "$TEMP_DIR" 2>/dev/null || true
    
    # Clean old backups (keep last 10)
    if [ -d "$BACKUP_DIR" ]; then
        ls -1t "$BACKUP_DIR" | tail -n +11 | head -20 | while read old_backup; do
            rm -rf "$BACKUP_DIR/$old_backup"
            log "üóëÔ∏è Removed old backup: $old_backup"
        done
    fi
    
    log "‚úÖ Cleanup completed"
}

# Main deployment logic
main() {
    local deployment_type="${1:-hub}"
    local module_name="$2"
    local module_subdir="$3"
    
    log "üöÄ Starting InfoTech.io deployment..."
    log "Type: $deployment_type"
    log "Module: ${module_name:-N/A}"
    log "Subdirectory: ${module_subdir:-N/A}"
    
    case "$deployment_type" in
        "hub")
            deploy_hub
            ;;
